import re
from pprint import pprint

def parse_markdown_content(content, mapping):
    # Main pattern to match markdown headers and their content
    pattern = r'\*\*(.*?)\*\*\n(.*?)(?=\n\*\*|\Z)'
    matches = re.findall(pattern, content, re.DOTALL)
    
    result = {}
    for header, content in matches:
        cleaned_content = content.strip()
        
        # Add to result if mapping exists
        if header in mapping:
            result[mapping[header]] = cleaned_content
    
    return result
